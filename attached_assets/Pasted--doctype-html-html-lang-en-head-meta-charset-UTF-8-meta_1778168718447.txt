      // ── Recently played ───────────────────────────────────────────────────
      const RECENT_KEY = "recentlyPlayed";
      const RECENT_MAX = 4;

      function getRecent() {
        try { return JSON.parse(localStorage.getItem(RECENT_KEY)) || []; }
        catch { return []; }
      }

      function addRecent(name) {
        let r = getRecent().filter(n => n !== name);
        r.unshift(name);
        if (r.length > RECENT_MAX) r = r.slice(0, RECENT_MAX);
        localStorage.setItem(RECENT_KEY, JSON.stringify(r));
        renderRecent();
      }

      function clearRecent() {
        localStorage.removeItem(RECENT_KEY);
        renderRecent();
      }

      function makeCard(game, i) {
        return `
          <button class="game-card" data-name="${game.name.toLowerCase()}" data-index="${i}">
            <div class="game-thumb">
              <img class="game-icon" src="${game.icon}" alt="${game.name}" loading="lazy">
              <div class="play-icon"><i class="fa-solid fa-play"></i></div>
            </div>
            <div class="game-name">${game.name}</div>
          </button>`;
      }

      function bindCard(card) {
        const game = games[card.dataset.index];
        card.addEventListener("click",      () => { selectCard(card); launchGame(game); });
        card.addEventListener("mouseenter", () => updateBG(game.bg));
        card.addEventListener("mouseleave", resetBG);
      }

      function renderRecent() {
        const names   = getRecent();
        const section = document.getElementById("recent-section");
        const row     = document.getElementById("recent-row");
        if (!names.length) { section.classList.remove("visible"); return; }
        const matched = names.map(n => {
          const i = games.findIndex(g => g.name === n);
          return i >= 0 ? { game: games[i], i } : null;
        }).filter(Boolean);
        if (!matched.length) { section.classList.remove("visible"); return; }
        row.innerHTML = matched.map(({ game, i }) => makeCard(game, i)).join("");
        row.querySelectorAll(".game-card").forEach(bindCard);
        section.classList.add("visible");
      }

      // ── Grid ──────────────────────────────────────────────────────────────
      function loadGames() {
        const grid     = document.getElementById("grid");
        const noResult = document.getElementById("no-results");
        document.getElementById("game-count").textContent = games.length + " games";

        grid.innerHTML = games.map((game, i) => makeCard(game, i)).join("") + noResult.outerHTML;

        document.getElementById("no-results").style.display = "none";

        document.querySelectorAll("#grid .game-card").forEach(bindCard);

        renderRecent();
      }

      function selectCard(el) {
        document.querySelectorAll(".game-card.selected").forEach(c => c.classList.remove("selected"));
        el.classList.add("selected");
      }

      function filterGames() {
        const q = document.getElementById("search").value.toLowerCase().trim();
        let visible = 0;
        document.querySelectorAll(".game-card").forEach(card => {
          const show = !q || card.dataset.name.includes(q);
          card.style.display = show ? "" : "none";
          if (show) visible++;
        });
        const nr = document.getElementById("no-results");
        if (nr) nr.style.display = visible === 0 ? "block" : "none";
      }

      // ── Background ────────────────────────────────────────────────────────
      function updateBG(bgUrl) {
        const bg = document.getElementById("dynamic-bg");
        bg.style.backgroundImage = `url('${bgUrl}')`;
        bg.classList.add("active");
      }

      function resetBG() {
        document.getElementById("dynamic-bg").classList.remove("active");
      }

      // ── Game launch ───────────────────────────────────────────────────────
      async function launchGame(game) {
        const { url, local, name } = game;
        const frame   = document.getElementById("game-frame");
        const overlay = document.getElementById("game-overlay");
        const loading = document.getElementById("game-loading");

        loading.classList.remove("hidden");
        document.getElementById("loading-label").textContent = "Loading " + name + "...";
        frame.src = "";
        overlay.classList.add("active");
        document.body.style.overflow = "hidden";
        addRecent(name);

        if (local) {
          frame.src = url;
          return;
        }

        try { await navigator.serviceWorker.ready; } catch (e) {}

        if (typeof __uv$config === "undefined") {
          alert("Proxy not ready. Please refresh the page.");
          return;
        }

        try {
          frame.src = __uv$config.prefix + __uv$config.encodeUrl(url);
        } catch (e) { console.error("URL encode error:", e); }
      }

      function hideLoading() {
        const l = document.getElementById("game-loading");
        if (l) l.classList.add("hidden");
      }

      function closeGame() {
        document.getElementById("game-overlay").classList.remove("active");
        document.getElementById("game-frame").src = "";
        document.getElementById("game-loading").classList.remove("hidden");
        document.body.style.overflow = "auto";
      }
